RHEV-M
1. Install 
# wget http://classroom.example.com/materials/rhevm.repo -P /etc/yum.repos.d/
# yum repolist
# yum -y install rhevm rhevm-dwm rhevm-reports
# engine-setup --generate-answer=/root/answers.txt

> Pure Ovirt
# yum -y install http://resources.ovirt.org/pub/yum-repo/ovirt-release35.rpm
# yum -y install ovirt-engine
# touch /etc/exports 
# systemctl start rpcbind nfs-server
# engine-setup --generate-answer=/root/answers.txt

> To use generated answers.txt file next time for automated installation, do
# engine-setup --config-apend=/root/answers.txt
> To disable nfs v4 when using ovirt
# vi /etc/sysconfig/nfs
-  add to the end
    NFS4_SUPPORT="no"

Then:
# mkdir /var/lib/exports/data 
# chown vdsm:kvm /var/lib/exports/data 
# vi /etc/exports.d/ovirt-engine-iso.exports
- Add    
    /var/lib/exports/iso    172.25.1.0/24(rw)
    /var/lib/exports/data   172.25.1.0/24(rw)

- Restart nfs server
# systemctl restart rpc-statd nfs-server
